\section{Algorithms}
And here, just an small example of an algorithm.

\RestyleAlgo{boxed} %Comment this line with '%' to change the aspect of the box. :-)
\begin{algorithm}[H]
\RestyleAlgo{boxruled}
\label{tourment-algorithm}
\SetAlgoLined
\DontPrintSemicolon % Some LaTeX compilers require you to use \dontprintsemicolon instead
%\KwIn{a complex network $G$.}
%\KwOut{the dendrogram with partitions of $G$.}
$Input: (Population_0, Tournment\_Size)$;\;
%$Population_0 \leftarrow Current\_Population;$\;
%$Tournment\_Size \leftarrow Tournment\_Size\_Parameter;$\;
$Population_1 = New\_Population;$\;
$Population_1 \leftarrow \{Indindividual_1,...,Individual_{Tournment\_Size}\};$\;
\For{$i =0; \leqslant Tournment\_Size$}{
	$Population_1[i] \leftarrow Population_0[random[0,Tournment\_Size]];$\;
}
$Output : Population_1.Best.$\;
\caption{Tournament selection.}
\end{algorithm}
